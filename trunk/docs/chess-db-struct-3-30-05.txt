  `game_id` bigint(20) unsigned NOT NULL auto_increment,
  `whose_turn` mediumint(8) unsigned NOT NULL default '0',
  `white_player` mediumint(8) unsigned NOT NULL default '0',

  `message_from` enum('black','white') default NULL,
  `date_created` datetime NOT NULL default '0000-00-00 00:00:00',
  `last_move` datetime NOT NULL default '0000-00-00 00:00:00',
  `winner` mediumint(8) default NULL,
  `experience_lost` int(4) default NULL,
  `experience_won` int(4) default NULL,
  `rated` enum('0','1') NOT NULL default '1',
  `reason` varchar(30) NOT NULL default '',
  `game_ok` enum('0','1') NOT NULL default '1',
  `time_black` int(11) NOT NULL default '0',
  `time_white` int(11) NOT NULL default '0',
  `flag_fall` enum('0','1') NOT NULL default '0',
  `time_limit` int(11) NOT NULL default '0',
  `tournament` bigint(10) NOT NULL default '0',
  `team_match` enum('0','1') NOT NULL default '0',
  `team` bigint(10) NOT NULL default '0',
  `post_game` enum('0','1') NOT NULL default '0',
  `thematic` int(6) NOT NULL default '0',
  `reminder` enum('0','1') NOT NULL default '0',
  PRIMARY KEY  (`game_id`),
  KEY `date` (`date_created`),
  KEY `lastMove` (`last_move`),
